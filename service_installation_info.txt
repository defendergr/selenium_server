#installation
sudo mv /etc/systemd/system/server_start.service

#reload
sudo systemctl daemon-reload
sudo systemctl enable server_start.service
sudo systemctl start server_start.service
sudo systemctl status server_start.service

#uninstall
sudo rm /etc/systemd/system/server_start.service
sudo systemctl daemon-reload

#view logs
sudo journalctl -u server_start.service -f -q
